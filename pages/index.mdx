import { useSSG } from 'nextra/ssg'
import Parser from "rss-parser";
import Link from "next/link";


export const getStaticProps = ({ params }) => {
    let parser = new Parser();
    return parser.parseURL("https://blog.openduct.net/feed.xml").then((res) => ({
        props: {
            ssg: {
                items: res.items,
            }
        },
        revalidate: 1,
    }))
}


export default () => {
    const { items } = useSSG()
    return (
        <>
            {items.map((item) => (
            <a
                key={item.link}
                href={item.link}
                target="_blank"
                rel="noopener noreferrer"
            >
                <div>{item.title}</div>
            </a>
            ))}
        </>
    )
}
